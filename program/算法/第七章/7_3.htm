3. 构造图的十字链表算法

Status CreateDG ( OLGraph &G ) {

// 采用十字链表存储表示，构造有向图 G（G.kind = DG）

scanf ( &G.vexnum, &G.arcnum, &IncInfo ); // IncInfo 为0则各弧不含其他信息

for ( i = 0; i < G.vexnum; ++i ) { // 构造表头向量

scanf ( &G.xlist [i].data ); // 输入顶点值

G.xlist[i].firstin = NULL; // 初始化指针

G.xlist[i].firstout = NULL; // 初始化指针

} // for 结束

for ( k = 0; k < G.arcnum; ++k ) { // 输入各弧并构造十字链表

scanf ( &v1, &v2 ); // 输入一条弧的始点和终点

i = LocateVex ( G, v1 ); // 确定 v1 在 G 中的位置

j = LocateVex ( G, v2 ); // 确定 v2 在 G 中的位置

while ( i < 1 || i > G.vexnum || j < 1 || j > G.vexnum) {

// 如果编号超出范围，重新输入弧的始点和终点，并确定它们在 G 中的位置

scanf ( &v1, &v2 );

i = LocateVex ( G, v1 );

j = LocateVex ( G, v2 );

} // while 结束

if ( ! p = ( ArcBox * ) malloc ( sizeof ( ArcBox ) ) )

exit ( OVERFLOW ) // 若没有足够空间则退出

p.tailvex = i; // 对弧结点的尾域赋值

p.headlvex = j; // 对弧结点的头域赋值

p.hlink = G.xlist [j].firstin; // 对弧结点的链域赋值

p.tlink = G.xlist [i].firstout; // 对弧结点的链域赋值

p.info = NULL // 对弧结点相关信息指针域赋值

G.xlist[j].firstin = G.xlist[i].firstout = p; // 完成在入弧和出弧链头的插入

if ( IncInfo ) Input ( *p->info ); // 如果弧含有相关信息，则输入

} // for 结束

return OK

} // CreateDG
