14. 在链式队列尾插入新元素算法

Status EnQueue ( LinkQueue &Q; QElemType e ) {

// 插入元素 e 为队列 Q 的新的队列尾元素；

p = ( QueuePtr ) malloc ( sizeof ( QNode ) );

if ( ! p ) exit ( OVERFLOW ); // 存储空间分配失败

p->data = e; // 将值 e 放入新结点的数据域

p->next = NULL; // 令新结点的指针域为空

Q.rear->next = p; // 将新结点插入到队列 Q 的尾

Q.rear = p; // 修改队列 Q 的队尾指针

return OK;

} // EnQueue
