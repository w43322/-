1. 直接插入排序算法

void InsertSort ( SqList &L ) {

// 对顺序表 L 作直接插入排序

for ( i = 2; i <= L,length; ++i )

if ( L.r[i].key < L.r[i-1].key ) { // 若“<”，则将 L.r[i] 插入有序子表

L.r[0] = L.r[i]; // 复制为哨兵

for ( j = i-1; L.r[0].key < L.r[j].key ); --j )

L.r[j+1] = L.r[j]; // 记录后移

L.r[j+1] = L.r[0]; // 插入到正确位置

} // if 结束

} // InsertSort
