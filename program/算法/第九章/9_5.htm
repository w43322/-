5. 直接选择排序算法

void SelectSort ( SqList &L ) {

// 对顺序表 L 作直接选择排序

for ( i = 1; i < L.length; ++i ) { // 选择第 i 个小的记录，并交换到位

k = i;

for ( j = i+1; j < L.length; ++j ) // 在 L.r[i..length] 中选择 key 最小的记录

if ( L.r[j].key < L.r[k].key ) k = j;

if ( i ! = k ) { // 与第 i 个记录交换

temp = L.r[i]; L.r[i ] = L.r[k]; L.r[k] = temp;

} // if 结束

} // for 结束

} // SelectSort
