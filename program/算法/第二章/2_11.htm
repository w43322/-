11. 例 2-1 写一算法。将两个有序表 A 和 B 归并成一个有序表 C，其中 A，B，C 均为 SqList 类型（即顺序表类型，为一结构体，含 length，listsize 和 elem 数组指针部分）的参数。

Status Merge_Sq ( SqList A, SqList B, SqList C ) {

// 将两个有序顺序表 A 和 B 合成一个有序顺序表 C

C.listsize = C.length = A.length + B.length; // 设置 C 表的存储容量和表长

C.elem = ( ElemType *) malloc ( C.listzise * sizeof ( ElemType ) )

if ( ! C.elem ) exit ( OVERFLOW ); // 存储空间分配失败

i = 0; j = 0; // i 和 j 是 A 和 B 的 elem 数组的下标

k = 0; // k 是 C 的 elem 数组的下标

while ( ( i <= A .length ) && ( j <= B.length ) ) // 归并

if ( A.elem [i] <= B.elem [j] ) {

C.elem [k] = A.elem [i];

i++; k++;

} // if 结束

else {

C.elem [k] = B.elem [j];

j++; k++;

} // else 结束

while ( i <= A.length ) { // 插入有序顺序表 A 的剩余段

C.elem [k] = A.elem [i];

i++; k++;

} // while 结束

while ( j <= B.length ) { // 插入有序顺序表 B 的剩余段

C.elem [k] = B.elem [j];

j++; k++;

} // while 结束

return OK;

} // Merge_Sq
