14. 例 2-4 写一算法。假设头指针为 La 和 Lb 的两个有序单链表分别为两个有序线性表 LA 和 LB 的存储结构。要求归并 La 和 Lb 得到一个有序单链表 Lc，并且 Lc 和 La 共用一个表头。

void MergeList_L ( LinkList &La, LinkList &Lb, LinkList &Lc, ) {

// 已知单链表 La 和 Lb 的元素按非递减排列，

// 归并 La 和 Lb 得到新的单链表 Lc ，Lc 的元素也按非递减排列

pa = La->next; pb = Lb->next;

Lc = pc = La; // 用 La 的头结点作为 Lc 的头结点

while ( pa && pb ) {

if ( pa->data <= pb->data ) { // 如果 pa->data≤pb->data

pc->next = pa;

pc = pa;

pa = pa->next;

} // if 结束

else { // 如果 pa->data > pb->data

pc->next = pb;

pc = pb;

pb = pb->next;

} // else 结束

} // while结束

pc->next = pa ? pa : pb; // 插入剩余段

free (Lb); // 释放 Lb 的头结点

} // MergeList_L
