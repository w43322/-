5. 单链表的取元素算法

Status GetElem_L ( LinkList L, int i, ElemType &e ) {

// L 为带头结点的单链表的头指针，当第 i 个元素存在时，其值赋给 e 并返回 OK，否则返回 ERROR

p = L->next; j = 1; // 初始化。p 指向链表第一个结点，j 为计数器

while ( p && j<i ) { p = p->next; ++j; } // 顺指针向后查直到 p 指向第 i 个元素或为空

if ( ! p || j>i ) return ERROR; // 第 i 个元素不存在

e = p->data; // 取第 i 个元素

return OK;

} // GetElem_L
