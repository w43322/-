2. 销毁数组算法

Status DestroyArray ( Array &A ) {

// 销毁数组 A

if ( ! A.base ) return ERROR;

free ( A.base ); // 释放 A.base 空间

A.base = NULL; // 令 A.base 指针为空

if ( ! A.bounds ) return ERROR;

free ( A.bounds ); // 释放 A.bounds 空间

A.bounds = NULL; // 令 A.bounds 指针为空

if ( ! A.constants ) return ERROR;

free ( A.constants ); // 释放 A.contants 空间

A.constants = NULL; // 令 A.constants 指针为空

return OK;

} // DestroyArray
