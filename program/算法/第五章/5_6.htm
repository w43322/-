6. 求矩阵转置算法

Status TransposeSMatrix ( TSMatrix M, TSMatrix &T ) {

// 采用三元组顺序表结构，求稀疏矩阵 M 的转置矩阵 T。在算法中：

// q 指示 T.data 中三元组的序号，p 指示 M.data 中三元组的序号，col 指示 M 的列号

T.mu = M.nu;

// 将稀疏矩阵 M 的列数值作为其转置矩阵 T 的行数值

T.nu = M.mu;

// 将稀疏矩阵 M 的行数值作为其转置矩阵 T 的列数值

T.tu = M.tu;

// 转置矩阵 T 与稀疏矩阵 M 的非零元个数相等

if ( T.tu ) {

q = 1;

for ( col = 1; col < = M.nu; ++col )

for ( p = 1; p < = M.tu; ++p )

if ( M.data[p].j = = col ) { // 进行转置

T.data[q].i = M.data[p].j; // M 的列域值成为其转置矩阵 T 的行域值

T.data[q].j = M.data[p].i; // M 的行域值成为其转置矩阵 T 的行域值

T.data[q].e = M.data[p].e; // 将 M 的非零元值赋给其转置矩阵 T

++q; // T.data 中三元组的序号加 1

} // if ( M.data ) 结束

} // if ( T.tu ) 结束

return OK;

} // TransposeSMatrix
