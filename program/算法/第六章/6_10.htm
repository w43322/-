10. 在二叉线索树中插入结点算法

Status Insert_Thr ( BiThrTree Thrt, BiThrTree t, TElemType x ) {

// 构造数据域值为 x 的新结点，将其插入到二叉线索树中作为结点 t 的右孩子，并返回 OK

if ( ! q = ( BiThrTree ) malloc ( sizeof ( BiThrNode ) ) ) exit ( OVERFLOW );

q->data = x; q->rchild = t->rchild; q->rtag = t->rtag;

q->lchild = t; q->ltag = Thread; t->rchild = q; t->rtag = Link;

if ( q->rtag = Link ) { Next_Thr ( q, p ); p->lchild = q; }

// 若 q 的右标志为 0，则寻找 q 的后继 p，并将 q 作为 p 的前驱

return OK

} // Insert_Thr
