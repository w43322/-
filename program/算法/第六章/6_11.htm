11. 遍历中序线性二叉树算法

Status InOrderTraverse_Thr ( BiThrTree Thrt, Status ( * Visit ) ( TElemType e ) ) {

// T 指向中序线索树头结点，头结点的左指针 lchild 指向根结点，遍历二叉线索树 T 的非递归算法

p = Thur->lchild; // p 指向根结点

while ( p != Thur ) { // 当空树或遍历结束时，p = = Thur

while ( p->ltag = = Link ) p = p->lchild; // 顺着 p 左链域寻找，直到结点 ltag 为 1。

if ( ! Visit ( p->data ) ) return ERROR; // 访问其左链域为空的结点

while ( p->rtag = = Thread && p->rchild ! = Thur ) { p = p->rchild; Visit ( p->data ); }

// 当 p 右标志为 1 且右指针 rchild 不为头结点时，访问后继

p = p->rchild;

} // while ( p != Thur ) 结束

return OK;

} // InOrderTraverse
