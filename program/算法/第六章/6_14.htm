Status Insert_Thr ( BiThrTree Thrt, BiThrTree t, TElemType e ) {

// 构造数据域值为 e 的新结点，将其插入到线索树中作为 t 的左孩子，并返回 OK

q->data = e;

q->lchild = t->lchild; q->ltag = t->ltag;

q->rchild = t; q->rtag = Thread;

t->lchild = q; t->ltag = Link;

if ( q->LTag = Link ) {

} // if 结束
