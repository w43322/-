8. 在二叉线索树中求结点的前驱算法

Status Prior_Thr ( BiThrTree t, &p ) {

// t 指向二叉线索树中某一结点，p 返回此结点的前驱；并返回 OK

p = t->lchild; // 将 t 的 lchild 域的值赋给 p

if ( p = Thrt ) return ERROR; // 若 p 指向头结点，则出错

if ( t->ltag = Link ) // 若 t 的 ltag 为 0

while ( p->rtag = Link ) { p = p->rchild; } // 顺着 t 左孩子的右链域寻找，直到 rtag 为 1

return OK;

} // Prior_Thr
