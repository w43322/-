9. 在二叉线索树中求结点的后继算法

Status Next_Thr ( BiThrTree t, &p ) {

// t 指向二叉线索树中某一结点，p 返回此结点的后继；并返回 OK

p = t->rchild; // 将 t 的 rchild 域的值赋给 p

if ( p = Thrt ) return ERROR; // 若 p 指向头结点，则出错

if ( t->rtag = Link ) // 若 t 的 rtag 为 0

while ( p->lrag = Link ) { p = p->lchild; } // 顺着 t 右孩子的左链域寻找，直到 ltag 为 1

return OK;

} // Next_Thr
