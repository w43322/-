1. 联接定长顺序存储的串算法

Status Concat ( SString &T, SString S1, SString S2 ) {

// 用 T 返回由 S1 和 S2 联接而成的新串。

// 如果没有截断，则返回 TRUE，否则返回 FALSE。

// 假定以下标为 0 的数组分量存放串实际长度。

if ( S1[0] + S2[0] <= MAXSTRLEN ) { // 没有截断

T[ 1.. S1[0] ] = S1[ 1.. S1[0] ];

T[ S1[0] + 1.. S1[0] + S2[0] ] = S2[ 1.. S2[0] ];

T[0] = S1[0] + S2[0];

uncut = TURE;

} // if 结束

else if ( S1[0] < MAXSTRLEN ) { // 截断 S2

T[ 1.. S1[0] ] = S1[ 1.. S1[0] ];

T[ S1[0] + 1.. MAXSTRLEN ] = S2[ 1.. MAXSTRLEN C S1[0] ];

T[0] = MAXSTRLEN;

uncut = FALSE;

} // else if 结束

else if ( S1[0] = MAXSTRLEN ) { // 截断（仅取 S1）

T[ 0.. MAXSTRLEN ] = S1[ 0 .. MAXSTRLEN ];

// 因为 T[0] = = S1[0] = = MAXSTRLEN

uncut = FALSE;

} // else if 结束

else { // 若 S1[0] > MAXSTRLEN，截断 S1

T[ 1.. MAXSTRLEN ] = S1[ 1.. MAXSTRLEN ];

T[0] = MAXSTRLEN;

uncut = FALSE;

} // else 结束

return uncut;

} // Concat
