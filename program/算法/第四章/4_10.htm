// 比较两个字符串 S1 和 S2：

// 如果 S1 > S2，则输出一个正数；如果 S1 = S2，则输出零；如果 S1 < S2，则输出一个负数。

// 字符串采用定长顺序存储表示。字符串以“\0”表示串值的终结。

for ( i = 0; S1[i] ¹ ¢ \0¢ && S2[i] ¹ ¢ \0¢ ; ++i )

if ( S1[i] != S2[i] ) return S1[i]-S2[i]; // 返回两个字符的 ASCII 码差值

for ( m = 0; S1[m] ¹ ¢ \0¢ ; ++m ); // 求 S1 串长，m = S1 串长

for ( n = 0; S2[n] ¹ ¢ \0¢ ; ++n ); // 求 S2 串长，n = S2 串长

} // Compare
