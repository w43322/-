2. 求定长顺序存储的子串算法

Status SubString ( SString &Sub, SString S, int pos, int len) {

// 用 Sub 返回串 S 的第 pos 个字符起长度为 len 的子串。其中，

// 1≤pos≤StrLength (S) 且 0≤len≤StrLength (S)Cpos+1。

// 假定以下标为 0 的数组分量存放串的实际长度。

if ( pos < 1 || pos > S[0] || len < 0 || len > S[0]-pos+1 ) return ERROR;

Sub[ 1.. len ] = S[ pos .. pos+len-1 ]; Sub[0] = len;

return OK;

} // SubString;
